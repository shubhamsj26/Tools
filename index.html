    <script>

        // Get references to DOM elements

        const gameBoard = document.getElementById('game-board');

        const gameStatus = document.getElementById('game-status');

        const resetButton = document.getElementById('reset-button');

        const cells = document.querySelectorAll('.board-cell'); // Get all board cells


        // Game state variables

        let board = ['', '', '', '', '', '', '', '', '']; // Represents the 3x3 grid

        let currentPlayer = 'X'; // 'X' or 'O'

        let gameActive = true; // True if the game is ongoing, false otherwise


        // Winning conditions (indices of cells that form a win)

        const winningConditions = [

            [0, 1, 2], // Top row

            [3, 4, 5], // Middle row

            [6, 7, 8], // Bottom row

            [0, 3, 6], // Left column

            [1, 4, 7], // Middle column

            [2, 5, 8], // Right column

            [0, 4, 8], // Diagonal from top-left

            [2, 4, 6]  // Diagonal from top-right

        ];


        /**

         * Initializes the game state and updates the UI.

         */

        function initializeGame() {

            board = ['', '', '', '', '', '', '', '', '']; // Reset board

            currentPlayer = 'X'; // Start with Player X

            gameActive = true; // Set game as active

            gameStatus.textContent = `Player ${currentPlayer}'s Turn`; // Update status text


            // Clear cell content and remove styling

            cells.forEach(cell => {

                cell.textContent = '';

                cell.classList.remove('x-color', 'o-color', 'filled', 'winning-cell');

            });

        }


        /**

         * Handles a click on a game board cell.

         * @param {Event} event The click event object.

         */

        function handleCellClick(event) {

            const clickedCell = event.target;

            const clickedCellIndex = Array.from(cells).indexOf(clickedCell); // Get index of clicked cell


            // If the cell is already filled or game is not active, do nothing

            if (board[clickedCellIndex] !== '' || !gameActive) {

                return;

            }


            // Update board state and UI

            board[clickedCellIndex] = currentPlayer;

            clickedCell.textContent = currentPlayer;

            clickedCell.classList.add('filled', `${currentPlayer.toLowerCase()}-color`); // Add filled and color class


            checkGameResult(); // Check for win or draw

        }


        /**

         * Checks if the current player has won the game.

         * @returns {boolean} True if the current player has won, false otherwise.

         */

        function checkWin() {

            let roundWon = false;

            for (let i = 0; i < winningConditions.length; i++) {

                const winCondition = winningConditions[i];

                let a = board[winCondition[0]];

                let b = board[winCondition[1]];

                let c = board[winCondition[2]];


                // If any cell in the condition is empty, continue

                if (a === '' || b === '' || c === '') {

                    continue;

                }

                // If all cells in the condition match the current player, it's a win

                if (a === b && b === c) {

                    roundWon = true;

                    // Highlight winning cells

                    winCondition.forEach(index => {

                        cells[index].classList.add('winning-cell');

                    });

                    break;

                }

            }

            return roundWon;

        }


        /**

         * Checks if the game is a draw (all cells filled, no winner).

         * @returns {boolean} True if the game is a draw, false otherwise.

         */

        function checkDraw() {

            return !board.includes(''); // If no empty cells, it's a draw

        }


        /**

         * Checks the game result (win or draw) and updates the game status.

         */

        function checkGameResult() {

            if (checkWin()) {

                gameStatus.textContent = `Player ${currentPlayer} Wins! 🎉`;

                gameActive = false; // End the game

                return;

            }


            if (checkDraw()) {

                gameStatus.textContent = `It's a Draw! 🤝`;

                gameActive = false; // End the game

                return;

            }


            // If no win or draw, switch to the next player

            currentPlayer = currentPlayer === 'X' ? 'O' : 'X';

            gameStatus.textContent = `Player ${currentPlayer}'s Turn`;

        }


        // Add event listeners

        cells.forEach(cell => cell.addEventListener('click', handleCellClick));

        resetButton.addEventListener('click', initializeGame);


        // Initialize the game when the page loads

        window.onload = initializeGame;

    </script>
